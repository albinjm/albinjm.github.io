mcas_volatile.ml Descriptor Types
Core data types — MCAS descriptor encoding
type status =
  | Active
  | Successful
  | Failed

type state =
  | Value of Obj.t
  | Word  of word_desc

and word_desc = {
  address   : cell;          (* target memory location *)
  old_value : Obj.t;
  new_value : Obj.t;
  parent    : mcas_desc;    (* back-pointer to owning MCAS *)
}

and mcas_desc = {
  status  : status Atomic.t;   (* single-word CAS target *)
  words   : word_desc array;
  retired : bool Atomic.t;
}

and cell = state Atomic.t     (* each ref slot holds Value or Word *)

type 'a ref = { id : int; cell : cell }
